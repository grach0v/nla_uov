def richardson_iter(A, b, x_0, dump=1, max_iter=100, eps=1e-6, save_hist=True):
    x = x_0.copy()
    
    if save_hist:
        hist_residual = []

    for _ in range(max_iter):
        res = b - A @ x
        res_norm = np.linalg.norm(res)

        if res_norm < eps:
            break

        if save_hist:
            hist_residual.append(res_norm)

        x += dump * res
    
    if save_hist:
        return x, hist_residual
    
    return x

def jacobi_iter(A, b, x_0, max_iter=100, eps=1e-6, save_hist=True):
    x = x_0.copy()
    
    if save_hist:
        hist_residual = []

    D = np.diag(A) # Note D is vector, not matrix 
    L_U = A - D

    for _ in range(max_iter):
        res = b - A @ x
        res_norm = np.linalg.norm(res)

        if res_norm < eps:
            break

        if save_hist:
            hist_residual.append(res_norm)

        x = (1 / D) * (b - (L_U @ x))
    
    if save_hist:
        return x, hist_residual
    
    return x

def gauss_siden_iter(A, b, x_0, max_iter=100, eps=1e-6, save_hist=True):
    x = x_0.copy()
    
    if save_hist:
        hist_residual = []

    L = np.tril(A, 0)
    U = np.triu(A, 1)
    L_inv = np.linalg.inv(L)

    for _ in range(max_iter):
        res = b - A @ x
        res_norm = np.linalg.norm(res)

        if res_norm < eps:
            break

        if save_hist:
            hist_residual.append(res_norm)

        x = L_inv @ (b - U @ x)

    if save_hist:
        return x, hist_residual
    
    return x
